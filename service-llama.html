Usage Example

Python PyTorch
inference.py 
import torch
from llama import LLaMA, ModelArgs

# Initialize model args
model_args = ModelArgs(
    dim=4096,
    n_layers=32,
    n_heads=32,
    vocab_size=32000
)

# Load pre-trained weights
model = LLaMA(model_args)
checkpoint = torch.load("/path/to/llama-7b.pth")
model.load_state_dict(checkpoint)

# Run inference
prompts = ["The future of AI is"]
results = model.generate(prompts, max_gen_len=64)

print(results[0])
