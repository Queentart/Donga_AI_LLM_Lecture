generate_content.py 
import google.generativeai as genai

genai.configure(api_key='YOUR_API_KEY')

model = genai.GenerativeModel('gemini-pro')

response = model.generate_content(
    "Explain quantum computing to a 5 year old."
)

print(response.text)
